INSERT INTO Dictionary(LevelID, Theme)
SELECT LevelID, тему
FROM Level
WHERE LevelName = название уровня

SELECT DictionaryID FROM Dictionary left join Level ON Dictionary.LevelID = Level.LevelID WHERE Theme = тема AND LevelName = уровень

INSERT INTO Words(LangID, Spell, Phonetic)
SELECT LangID, 'слово', 'транскрипция'
FROM Languages
WHERE LangName = язык слова

SELECT WordID FROM Words WHERE Spell = 'слово' AND Phonetic = 'транскрипция'
